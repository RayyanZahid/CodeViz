    export const SupportedLanguage = {
      TS: 'ts',
      TSX: 'tsx',
      JS: 'js',
      JSX: 'jsx',
      PY: 'py',
    } as const;
    export type SupportedLanguage = typeof SupportedLanguage[keyof typeof SupportedLanguage];

    export interface ParseTask {
      filePath: string;            // project-relative path
      source: string;              // file content
      language: SupportedLanguage;
      sequenceId: number;          // monotonic per-result ID
    }

    export interface ImportInfo {
      source: string;              // the import specifier (e.g. './auth.service')
      isTypeOnly: boolean;         // for TS type-only imports
    }

    export interface ExportInfo {
      name: string;                // exported name (e.g. 'AuthService', 'default')
      isDefault: boolean;
      isTypeOnly: boolean;         // for TS type-only exports
    }

    export interface CallInfo {

      isTopLevel: boolean;         // direct call vs nested
    }

    export interface ParseResult {

      language: SupportedLanguage;

    }

      type: 'removed';

  <name>Task 2: Implement FileWatcher with chokidar v5, debounce, and batch emission</name>

  <action>
    Create `packages/server/src/watcher/FileWatcher.ts` implementing the file watcher:

    **Class: FileWatcher**
    - Constructor takes: `watchRoot: string` (absolute path), `onBatch: (batch: FileWatchBatch) => void`, optional `options: { debounceMs?: number, customIgnores?: string[] }`
    - Uses chokidar v5 `watch()` — per locked decision from CONTEXT.md:
      - `followSymlinks: false` (locked: ignore symlinks)
      - `ignoreInitial: false` (emit 'add' for existing files on startup for initial scan)
      - `ignored` function that filters:
        1. Directories in IGNORED_DIRS set: `node_modules`, `.git`, `dist`, `build`, `coverage`, `.next`, `__pycache__`, `.pytest_cache`, `.venv`, `venv`
        2. Files not in PARSEABLE_EXTENSIONS set: `.ts`, `.tsx`, `.js`, `.jsx`, `.py`
        3. Config files in IGNORED_FILES set: `package.json`, `tsconfig.json`, `tsconfig.base.json`, `.env`, `.gitignore`
        4. Any patterns from `customIgnores` (user override via .archlens.json — future)
    - Extension filter at watcher level per locked decision — non-parseable files never fire events
    - Listen on `'add'`, `'change'`, `'unlink'` events (WATCH-04: create, modify, delete)

    **Debounce + Coalesce pattern (WATCH-02, WATCH-03):**
    - Maintain `private pending: Map<string, WatchEvent>` — keyed by relativePath
    - On each event: compute `relativePath = path.relative(watchRoot, absolutePath).replace(/\\\\/g, '/')`, store in pending (last event type for same path wins = coalescing)
    - Single global setTimeout at 200ms (WATCH-02: 200ms debounce window)
    - On flush: convert Map values to array, assign monotonic sequenceStart, clear Map, call `onBatch(batch)`

    **Monotonic sequence counter:**
    - Private `sequenceCounter: number = 0` — incremented per event in the batch
    - `batch.sequenceStart` is the first sequence number in the batch

    **Lifecycle methods:**
    - `start(): Promise<void>` — starts watching, returns when chokidar emits 'ready'
    - `stop(): Promise<void>` — calls `watcher.close()`, clears any pending timer
    - `isWatching(): boolean` — returns current state

    **Error handling:**
    - Listen on chokidar `'error'` event, log via console.error (will be replaced by pino in pipeline integration)

    Use ESM imports: `import { watch, type FSWatcher } from 'chokidar';`
    Use `import path from 'node:path';`
    Import types from `@archlens/shared` (WatchEvent, FileWatchBatch, FileEventType).
  </action>

    - Manual smoke test (optional, will be validated in Plan 03 integration): create a temp directory, instantiate FileWatcher, touch a file, observe batch callback fires
