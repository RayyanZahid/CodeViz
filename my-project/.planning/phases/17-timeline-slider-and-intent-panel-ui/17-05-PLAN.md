**PlaybackController.ts** — Pure class (no React dependencies) following the ViewportController pattern from ArchCanvas.tsx.

```typescript
const BASE_INTERVAL_MS = 1000; // 1 snapshot per second at 1x speed

export class PlaybackController {
  private timer: ReturnType<typeof setInterval> | null = null;

  start(speed: number, onTick: () => Promise<boolean>): void {
    // onTick returns true if playback should continue, false if at live edge
    this.stop();
    const interval = BASE_INTERVAL_MS / speed;
    this.timer = setInterval(async () => {
      const shouldContinue = await onTick();
      if (!shouldContinue) {
        this.stop();
      }
    }, interval);
  }

  stop(): void {
    if (this.timer !== null) {
      clearInterval(this.timer);
      this.timer = null;
    }
  }

  /** Restart with a new speed (call when speed changes during playback) */
  restart(speed: number, onTick: () => Promise<boolean>): void {
    this.start(speed, onTick);
  }
}
```

   ```typescript
   useEffect(() => {
     function handleKeyDown(e: KeyboardEvent) {
       // Guard: only handle if not typing in an input/textarea
       const target = e.target as HTMLElement;
       if (target.tagName === 'INPUT' || target.tagName === 'TEXTAREA') return;

       const { snapshots } = replayStore.getState();
       if (snapshots.length === 0) return;

       switch (e.key) {
         case ' ': { // Space = play/pause
           e.preventDefault();
           const { isPlaying } = replayStore.getState();
           if (isPlaying) {
             replayStore.getState().setIsPlaying(false);
           } else {
             // If not in replay, enter at first snapshot
             if (!replayStore.getState().isReplay) {
               replayStore.getState().setCurrentSnapshotIndex(0);
               void loadSnapshotAndEnterReplay(snapshots[0].id).then(() => {
                 replayStore.getState().setIsPlaying(true);
               });
             } else {
               replayStore.getState().setIsPlaying(true);
             }
           }
           break;
         }
         case 'ArrowRight': { // Step forward
           e.preventDefault();
           const { currentSnapshotIndex: idx, isReplay: inReplay } = replayStore.getState();
           if (!inReplay && snapshots.length > 0) {
             replayStore.getState().setCurrentSnapshotIndex(0);
             void loadSnapshotAndEnterReplay(snapshots[0].id);
           } else if (idx < snapshots.length - 1) {
             const next = idx + 1;
             replayStore.getState().setCurrentSnapshotIndex(next);
             void loadSnapshotAndEnterReplay(snapshots[next].id);
           }
           break;
         }
         case 'ArrowLeft': { // Step backward
           e.preventDefault();
           const { currentSnapshotIndex: idx2, isReplay: inReplay2 } = replayStore.getState();
           if (inReplay2 && idx2 > 0) {
             const prev = idx2 - 1;
             replayStore.getState().setCurrentSnapshotIndex(prev);
             void loadSnapshotAndEnterReplay(snapshots[prev].id);
           }
           break;
         }
         case '+':
         case '=': { // Increase speed
           e.preventDefault();
           const speeds: Array<0.5 | 1 | 2 | 4> = [0.5, 1, 2, 4];
           const currentSpeed = replayStore.getState().playbackSpeed;
           const idx3 = speeds.indexOf(currentSpeed);
           const nextSpeed = speeds[Math.min(idx3 + 1, speeds.length - 1)];
           replayStore.getState().setPlaybackSpeed(nextSpeed);
           break;
         }
         case '-': { // Decrease speed
           e.preventDefault();
           const speeds: Array<0.5 | 1 | 2 | 4> = [0.5, 1, 2, 4];
           const currentSpeed = replayStore.getState().playbackSpeed;
           const idx4 = speeds.indexOf(currentSpeed);
           const prevSpeed = speeds[Math.max(idx4 - 1, 0)];
           replayStore.getState().setPlaybackSpeed(prevSpeed);
           break;
         }
       }
     }

     document.addEventListener('keydown', handleKeyDown);
     return () => document.removeEventListener('keydown', handleKeyDown);
   }, [onExitReplay]);
   ```

Note: The Space key handler must check `e.target` to avoid conflicts with browser defaults when focused on inputs.

When `diffBaseSnapshotId` changes from null to a value (or changes value):
1. Fetch the diff base snapshot: `const baseRes = await fetch('/api/snapshot/' + diffBaseSnapshotId)`
2. Parse the response to get base snapshot nodes
3. Compute diff sets:
   - `added`: node IDs in current replay snapshot but NOT in base → green
   - `removed`: node IDs in base but NOT in current replay snapshot → red
   - `changed`: node IDs in both but with different properties (zone, fileList length, edge counts) → amber
4. Restore any previous diff tint: `restoreDiffTint(nodeRenderer, diffTintedFills)`
5. Apply new diff tint: `diffTintedFills = applyDiffTint(nodeRenderer, added, removed, changed)`
6. Redraw graph layer

When `diffBaseSnapshotId` changes from a value to null (diff cleared):
1. Restore diff tint: `restoreDiffTint(nodeRenderer, diffTintedFills)`
2. Re-apply the blue replay tint (since we're still in replay mode)
3. Redraw graph layer

When `isReplay` changes to false (exit replay):
1. Also clear any diff tint: `restoreDiffTint(nodeRenderer, diffTintedFills)`
2. Reset `currentDiffBase = null`
